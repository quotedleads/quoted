Fleet Cold Email — Apps Script
Paste into Google Sheets → Extensions → Apps Script. Sends 20 emails/day automatically.
⚙️ Setup: Open your Fleet Tracker Google Sheet → Extensions → Apps Script → delete existing code → paste this → Save → Triggers → Add Trigger → sendInitialOutreach → Time-driven → Daily → 9am. Add sendFollowUps → Daily → 10am.

const GSD_CONFIG = {
  yourName:   "Christ Pierre",
  phone:      "(732) 430-6468",
  email:      "Goldservicedetailing@gmail.com",
  website:    "www.goldservicedetailing.com",
  sheetName:  "Sheet1",
  cols: { company:1, category:2, city:3, contactName:8, status:9, contactEmail:6, notes:11, sentDate:12, followup1:13, followup2:14 }
};

function sendInitialOutreach() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(GSD_CONFIG.sheetName);
  var rows = sheet.getDataRange().getValues();
  var sent = 0;
  for (var i = 1; i < rows.length; i++) {
    var row = rows[i];
    if (row[GSD_CONFIG.cols.status] !== "To Contact") continue;
    var email = row[GSD_CONFIG.cols.contactEmail];
    if (!email || row[GSD_CONFIG.cols.sentDate] || !isValidEmail(email)) continue;
    var company = row[GSD_CONFIG.cols.company];
    var contact = row[GSD_CONFIG.cols.contactName] || "Team";
    var city = row[GSD_CONFIG.cols.city] || "";
    var category = row[GSD_CONFIG.cols.category] || "";
    var notes = row[GSD_CONFIG.cols.notes] || "";
    var tpl = pickTemplate(category, notes);
    var subject = tpl.subject.replace("[Company Name]", company);
    var body = tpl.body.replace(/\[Company Name\]/g, company).replace(/\[Contact Name.*?\]/g, contact.split("/")[0].trim()).replace(/\[City.*?\]/g, city);
    GmailApp.sendEmail(email, subject, body, { name: "Gold Service Detailing", replyTo: GSD_CONFIG.email });
    sheet.getRange(i+1, GSD_CONFIG.cols.sentDate+1).setValue(new Date());
    sheet.getRange(i+1, GSD_CONFIG.cols.status+1).setValue("Contacted");
    sent++;
    Utilities.sleep(2000);
    if (sent >= 20) break;
  }
  Logger.log("Sent " + sent + " emails.");
}

function sendFollowUps() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(GSD_CONFIG.sheetName);
  var rows = sheet.getDataRange().getValues();
  var now = new Date();
  for (var i = 1; i < rows.length; i++) {
    var row = rows[i];
    var email = row[GSD_CONFIG.cols.contactEmail];
    var sentDate = row[GSD_CONFIG.cols.sentDate];
    if (!email || !sentDate || !isValidEmail(email)) continue;
    var status = row[GSD_CONFIG.cols.status];
    if (["Meeting Set","Closed Won","Closed Lost"].includes(status)) continue;
    var days = Math.floor((now - new Date(sentDate)) / 86400000);
    var company = row[GSD_CONFIG.cols.company];
    var contact = (row[GSD_CONFIG.cols.contactName] || "Team").split("/")[0].trim();
    if (days >= 3 && !row[GSD_CONFIG.cols.followup1]) {
      GmailApp.sendEmail(email, "Re: Fleet Detailing for " + company + " — Quick Follow-Up", followUp1(company, contact), { name: "Gold Service Detailing", replyTo: GSD_CONFIG.email });
      sheet.getRange(i+1, GSD_CONFIG.cols.followup1+1).setValue(new Date());
      Utilities.sleep(2000);
    }
    if (days >= 7 && row[GSD_CONFIG.cols.followup1] && !row[GSD_CONFIG.cols.followup2]) {
      GmailApp.sendEmail(email, "Last note — Fleet Detailing for " + company, followUp2(company, contact), { name: "Gold Service Detailing", replyTo: GSD_CONFIG.email });
      sheet.getRange(i+1, GSD_CONFIG.cols.followup2+1).setValue(new Date());
      Utilities.sleep(2000);
    }
  }
}

function isValidEmail(e) { return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(e); }

function pickTemplate(cat, notes) {
  cat = cat.toLowerCase(); notes = notes.toLowerCase();
  if (cat.includes("cold") || cat.includes("reefer")) return coldTemplate();
  if (cat.includes("moving")) return movingTemplate();
  if (cat.includes("delivery") || cat.includes("courier")) return deliveryTemplate();
  return truckingTemplate();
}

function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data = JSON.parse(e.postData.contents);
    sheet.appendRow([data.timestamp||new Date().toLocaleString(), data.firstName||"", data.lastName||"", data.phone||"", data.email||"", data.vehicle||"", data.package||"", data.addon||"", data.dateLocation||"", data.notes||""]);
    if (data.email) sendConfirmation(data);
    sendOwnerAlert(data);
    return ContentService.createTextOutput(JSON.stringify({result:"success"})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({result:"error",error:err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}

function onFormSubmit(e) {
  if (!e||!e.values) return;
  var row = e.values;
  var data = {timestamp:row[0],firstName:row[1],lastName:row[2],phone:row[3],email:row[4],vehicle:row[5],package:row[6],addon:row[7],dateLocation:row[8],notes:row[9]};
  if (data.email) sendConfirmation(data);
  sendOwnerAlert(data);
}

Review Request Automation
⭐ Setup: In your Google Sheet, create a tab called "Completed Jobs" with columns: Date | Client Name | Phone | Email | Sent Review. Set a daily trigger on sendReviewRequests to run at 10am. It automatically emails customers 24hrs after their job asking for a Google review.
Fires 24 hours after each completed job. Paste your Google Review link inside the script.
// ── REVIEW REQUEST AUTOMATION ──
// Set daily trigger: Extensions → Apps Script → Triggers → sendReviewRequests → Daily → 10am

const REVIEW_LINK = "https://g.page/r/YOUR_GOOGLE_REVIEW_LINK/review"; // ← replace this

function sendReviewRequests() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Completed Jobs");
  if (!sheet) { Logger.log("Create a 'Completed Jobs' tab first"); return; }
  var rows = sheet.getDataRange().getValues();
  var yesterday = new Date();
  yesterday.setDate(yesterday.getDate() - 1);
  var yStr = yesterday.toLocaleDateString();
  var sent = 0;
  for (var i = 1; i < rows.length; i++) {
    var row = rows[i];
    var jobDate = new Date(row[0]).toLocaleDateString();
    var clientName = row[1] || "there";
    var email = row[3];
    var alreadySent = row[4];
    if (jobDate !== yStr || alreadySent || !email) continue;
    var firstName = clientName.split(" ")[0];
    var subject = "How was your detail? ⭐ — Gold Service Detailing";
    var body = "Hi " + firstName + ",\n\nThank you for choosing Gold Service Detailing! We hope your vehicle is looking incredible.\n\nIf you have 30 seconds, a quick Google review means the world to us — it helps us grow and serve more customers like you:\n\n⭐ Leave a Review: " + REVIEW_LINK + "\n\nThank you!\n\n— Christ Pierre\nGold Service Detailing\n📞 (732) 430-6468\n🌐 www.goldservicedetailing.com";
    GmailApp.sendEmail(email, subject, body, {name:"Gold Service Detailing", replyTo:"Goldservicedetailing@gmail.com"});
    sheet.getRange(i+1, 5).setValue("Sent " + new Date().toLocaleDateString());
    sent++;
    Utilities.sleep(1500);
  }
  Logger.log("Review requests sent: " + sent);
}
